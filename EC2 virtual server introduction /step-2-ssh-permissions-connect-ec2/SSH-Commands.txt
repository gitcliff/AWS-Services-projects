# ------------------------------
# ▶️ HOW TO RUN / EXECUTE THIS CODE IN POWERSHELL (Windows Native)
# ------------------------------

# Move the key from Windows Downloads to your .ssh directory  
Move-Item -Path "$env:USERPROFILE\Downloads\docker-server.pem" -Destination "$env:USERPROFILE\.ssh"  
# ✅ What it does: Moves the private key file to the .ssh folder in your user profile  
# 🧠 Why it matters: Organizes SSH keys in the expected location for SSH clients on Windows  

# Check the key file details and confirm presence  
Get-ChildItem -Path "$env:USERPROFILE\.ssh\docker-server.pem" | Format-List  
# ✅ What it does: Shows detailed information about the key file  
# 🧠 Why it matters: Confirms that the key file is present and ready for permission changes  

# Set permissions on the key file to read-only for your user account  
icacls "$env:USERPROFILE\.ssh\docker-server.pem" /inheritance:r /grant:r "$($env:USERNAME):(R)"  
# ✅ What it does: Removes inherited permissions and grants only read access to your user  
# 🧠 Why it matters: Enforces strict file permissions to satisfy SSH client security requirements  

# Connect to EC2 instance using the private key  
ssh -i $env:USERPROFILE\.ssh\docker-server.pem ec2-user@<PUBLIC_IP_ADDRESS>  
# ✅ What it does: Uses the private key to open an SSH session to your EC2 instance  
# 🧠 Why it matters: Securely manages your remote AWS server for administration or deployment
